                    const osc = ctx.createOscillator();
                    const gain = ctx.createGain();
                    osc.connect(gain); gain.connect(ctx.destination);
                    osc.type = 'sine'; osc.frequency.setValueAtTime(600, ctx.currentTime);
                    osc.frequency.exponentialRampToValueAtTime(300, ctx.currentTime + 0.03);
                    gain.gain.setValueAtTime(0.05, ctx.currentTime);
                    gain.gain.exponentialRampToValueAtTime(0.01, ctx.currentTime + 0.03);
                    osc.start(); osc.stop(ctx.currentTime + 0.03);

'bg-cyan-500/20 text-cyan-300' : 'bg-white/10 text-white/60'}`}>{renderIdx + 1}</span>
                                            <span className={`text-sm sm:text-lg leading-tight sm:leading-normal ${isSelected ? 'text-white' : 'text-white/80'}`}>{optText}</span>
                                        </button>
                                    );
                                })}
                            </div>

                            {/* Always Active Main Action Button */}
                            <div className="flex justify-end relative z-10 mt-auto">
                                <button 
                                    onClick={handleMainAction}
                                    className="w-full sm:w-auto px-8 sm:px-10 py-3 sm:py-4 rounded-xl sm:rounded-2xl font-bold text-base sm:text-lg flex items-center justify-center gap-3 transition-all duration-300 bg-cyan-500 text-slate-900 shadow-[0_0_25px_rgba(34,211,238,0.5)] hover:bg-cyan-400 hover:shadow-[0_0_35px_rgba(34,211,238,0.7)] hover:-translate-y-1"
                                >
                                    {mainActionText}
                                    <LucideIcon name={mainActionText === (ui.reviewBtn || T.en.reviewBtn) ? "clipboard-check" : "arrow-right"} className="w-5 h-5" />
                                </button>
                            </div>
                        </div>

                        {/* Sticky Sidebar */}
                        <div className="w-full lg:w-[380px] flex flex-col gap-6 sticky top-24 self-start">
                            <div className="glass-panel-card p-6 sm:p-8">
                                <h3 className="text-xs sm:text-sm text-white/50 uppercase tracking-widest mb-4 sm:mb-6 font-semibold">{ui.sections}</h3>
                                <div className="flex flex-col gap-3 sm:gap-4">
                                    {translatedSections.map((section, idx) => {
                                        const isActive = currentSectionIdx === idx;
                                        return (
                                            <button key={section.id} onClick={() => { playSound('click'); setCurrentSectionIdx(idx); setCurrentQuestionIdx(0); }} className={`flex items-center gap-3 sm:gap-4 p-2.5 sm:p-3 rounded-xl sm:rounded-2xl transition-all text-left ${isActive ? 'bg-white/15 border border-white/20 shadow-md' : 'hover:bg-white/5 border border-transparent'}`}>
                                                <div className={`p-2 rounded-lg sm:rounded-xl ${isActive ? 'bg-cyan-500/20 text-cyan-300' : 'bg-black/20 text-white/50'}`}>
                                                    <LucideIcon name={section.icon} className="w-4 h-4 sm:w-5 sm:h-5" />
                                                </div>
                                                <span className={`text-xs sm:text-sm leading-tight ${isActive ? 'text-white font-medium' : 'text-white/60'}`}>{section.title}</span>
                                            </button>
                                        )
                                    })}
                                </div>
                            </div>

                            <div className="glass-panel-card p-6 sm:p-8">
                                <h3 className="text-lg sm:text-xl font-medium mb-4 sm:mb-6">{ui.quizProgress}</h3>
                                <div className="space-y-3 sm:space-y-4">
                                    <div className="flex justify-between items-center text-xs sm:text-sm">
                                        <span className="text-white/60">{ui.questionsDone}:</span>
                                        <span className="font-mono text-sm sm:text-base">{answeredCount} / {totalQuestions}</span>
                                    </div>
                                    <div className="flex justify-between items-center text-xs sm:text-sm">
                                        <span className="text-white/60">{ui.timeMode}:</span>
                                        <span className="font-mono text-sm sm:text-base text-cyan-300">{mode === 'timed' ? formatTime(timeRemaining) : ui.zenMode}</span>
                                    </div>
                                    <button 
                                        onClick={() => { playSound('click'); setAppState('review'); }}
                                        className="w-full mt-2 sm:mt-4 py-2.5 sm:py-3 rounded-lg sm:rounded-xl border border-white/20 bg-white/5 hover:bg-white/10 transition-colors text-xs sm:text-sm font-medium flex items-center justify-center gap-2"
                                    >
                                        <LucideIcon name="clipboard-list" className="w-4 h-4" />
                                        {ui.reviewBtn || T.en.reviewBtn}
                                    </button>
                                </div>
                            </div>
                        </div>
                    </div>
                    {exitModalJSX}
                    {submitModalJSX}
                </div>
            );
        }

        const root = ReactDOM.createRoot(document.getElementById('root'));
        root.render(<App />);
    </script>
